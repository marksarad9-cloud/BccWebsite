<script>
function showPage(pageId) {
document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
document.getElementById(pageId).classList.add('active');
}

function showCourse(course) {
let title = document.getElementById("course-title");
let desc = document.getElementById("course-description");
let box = document.getElementById("course-details");

if (course === "bsit") {
title.innerText = "BS Information Technology (BSIT)";
desc.innerText = "Focuses on programming, web development, networking, cybersecurity, and system development.";
}
else if (course === "bshm") {
title.innerText = "BS Hospitality Management";
desc.innerText = "Covers hotel, restaurant, and tourism operations.";
}
else if (course === "bsba") {
title.innerText = "BS Business Administration";
desc.innerText = "Focuses on marketing, finance, and management.";
}
else if (course === "criminology") {
title.innerText = "BS Criminology";
desc.innerText = "Focuses on crime detection and law enforcement.";
}
else if (course === "education") {
title.innerText = "BS Education";
desc.innerText = "Prepares future teachers and educators.";
}
else if (course === "tourism") {
title.innerText = "BS Tourism Management";
desc.innerText = "Focuses on travel and tourism industry.";
}
else if (course === "accounting") {
title.innerText = "BS Accounting Information System";
desc.innerText = "Combines accounting with IT systems.";
}
else if (course === "entrepreneurship") {
title.innerText = "BS Entrepreneurship";
desc.innerText = "Focuses on business startups and innovation.";
}

box.style.display = "block";
}
</script>
